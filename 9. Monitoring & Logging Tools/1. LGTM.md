## ✅ PRODUCTION WINDOWS CLIENT CONFIG

```yaml
receivers:
  # Receive traces from .NET app
  otlp:
    protocols:
      grpc:
      http:

  # Windows system metrics
  windowsperfcounters:
    collection_interval: 60s
    perfcounters:
      - object: "Processor"
        instances: ["_Total"]
        counters:
          - name: "% Processor Time"
      - object: "Memory"
        counters:
          - name: "Available MBytes"
      - object: "LogicalDisk"
        instances: ["C:"]
        counters:
          - name: "% Free Space"

  # Windows Event Logs
  windowseventlog:
    channel: System

  # IIS logs
  filelog:
    include:
      - C:\inetpub\logs\LogFiles\W3SVC1\*.log
    start_at: beginning

processors:
  batch:
  memory_limiter:
    check_interval: 1s
    limit_mib: 256

exporters:
  # Send metrics to Prometheus
  prometheusremotewrite:
    endpoint: "http://10.10.20.15:9090/api/v1/write"

  # Send logs to Loki
  loki:
    endpoint: "http://10.10.20.15:3100/loki/api/v1/push"

  # Send traces to Tempo
  otlp/tempo:
    endpoint: "10.10.20.15:4317"
    tls:
      insecure: true

service:
  pipelines:
    metrics:
      receivers: [windowsperfcounters]
      processors: [memory_limiter, batch]
      exporters: [prometheusremotewrite]

    logs:
      receivers: [windowseventlog, filelog]
      processors: [batch]
      exporters: [loki]

    traces:
      receivers: [otlp]
      processors: [batch]
      exporters: [otlp/tempo]
```

## Program.cs Production Example

```csharp
builder.Services.AddOpenTelemetry()
    .WithTracing(tracer =>
    {
        tracer
            .AddAspNetCoreInstrumentation()
            .AddHttpClientInstrumentation()
            .AddOtlpExporter(opt =>
            {
                opt.Endpoint = new Uri("http://localhost:4317");
                opt.Protocol = OtlpExportProtocol.Grpc;
            });
    });
```
